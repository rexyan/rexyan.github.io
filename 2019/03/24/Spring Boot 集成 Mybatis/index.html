Maven 构建 Spring Boot

之前使用 IDEA 自动创建 Spring boot 程序，简直就是开箱即用。Maven 生成的 Spring boot 的项目，生成的项目和之前 IDEA 创建的相比，少了 application.properties 和 主入口 main 文件，和 pom.xml 里面的 Spring boot 依赖等。

添加依赖

Spring boot 依赖

<!-- 继承 Spring boot 的父级项目依赖，让其成为一个 Spring boot 项目 -->
<parent>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-parent</artifactId>
    <version>2.1.3.RELEASE</version>
    <relativePath/> <!-- lookup parent from repository -->
</parent>

<dependencies>
    <!--Spring boot 开发 web 项目的起步依赖，让其成为一个 web 项目 -->
    <dependency>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-web</artifactId>
    </dependency>
</dependencies>

数据库依赖

<!-- 加载  mybatis 整合 Spring Boot -->
<dependency>
    <groupId>org.mybatis.spring.boot</groupId>
    <artifactId>mybatis-spring-boot-starter</artifactId>
</dependency>

<!-- 加载  mysql JDBC 驱动包 -->
<dependency>
    <groupId>mysql</groupId>
    <artifactId>mysql-connector-java</artifactId>
</dependency>

配置 Mybatis Mapper.xml 所在位置

Maven 构建的项目没有核心配置文件 application.properties 文件，需要我们手工创建。并且在里面配置 Mybatis Mapper.xml 所在位置和数据库的连接信息

# 配置 mybatis 配置文件位置
mybatis.mapper-locations=classpath:me/yanrs/springboot/mapper*.xml

# 配置数据库连接信息
spring.datasource.username=root
spring.datasource.password=123456
spring.datasource.driver-class-name=com.mysql.jdbc.Driver
spring.datasource.url=jdbc:mysql://127.0.0.1:3306/springboot?useUnicode=true&characterEncoding=utf8&useSSL=false

使用 GeneratorMapper.xml 反向生成 mapper 和 model

在根目录下新建 GeneratorMapper.xml，GeneratorMapper.xml 会根据数据库中已经有了的信息，反向生成 mapper 和 model。我们要修改里面的数据库信息，表信息和包路径信息。

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE generatorConfiguration
        PUBLIC "-//mybatis.org//DTD MyBatis Generator Configuration 1.0//EN"
        "http://mybatis.org/dtd/mybatis-generator-config_1_0.dtd">

<generatorConfiguration>
    <!--指定连接数据库的JDBC驱动包位置-->
    <classPathEntry location="/Users/rex/Desktop/jar包/dbcp连接池/mysql-connector-java-5.1.5-bin.jar"/>

    <!--配置table表格信息内容体，targetRuntime指定采用Mybatis3版本-->
    <context id="tables" targetRuntime="Mybatis3">

        <!--抑制生成注释，由于生成注释是英文，可以不让生成-->
        <commentGenerator>
            <property name="suppressAllComments" value="true"/>
        </commentGenerator>

        <!--配置数据库连接信息-->
        <jdbcConnection driverClass="com.mysql.jdbc.Driver"
                        connectionURL="jdbc:mysql://127.0.0.1:3306/springboot?serverTimezone=GMT"
                        userId="root"
                        password="123456">
        </jdbcConnection>

        <!--生成model类，targetPackage指定model类包名，targetProject指定生成model放在哪个工程下-->
        <javaModelGenerator targetPackage="me.yanrs.springboot.model" targetProject="src/main/java">
            <property name="enableSubPackages" value="false"/>
            <property name="trimString" value="false"/>
        </javaModelGenerator>

        <!--生成mybatis的Mapper.xml文件，targetPackage指定Mapper.xml包名，targetProject指定生成Mapper.xml放在哪个工程下-->
        <sqlMapGenerator targetPackage="me.yanrs.springboot.mapper" targetProject="src/main/java">
            <property name="enableSubPackages" value="false"/>
        </sqlMapGenerator>

        <!--生成mybatis的Mapper接口类文件，targetPackage指定Mapper接口类的包名，targetProject指定生成Mapper接口放在哪个工程下-->
        <javaClientGenerator type="XMLMAPPER" targetPackage="me.yanrs.springboot.mapper" targetProject="src/main/java">
            <property name="enableSubPackages" value="false"/>
        </javaClientGenerator>

        <!--数据库表名及对应java模型类名-->
        <table tableName="student"
               domainObjectName="Student"
               enableCountByExample="false"
               enableUpdateByExample="false"
               enableDeleteByExample="false"
               enableSelectByExample="false"
               selectByExampleQueryId="false"/>
    </context>
</generatorConfiguration>

在 pom.xml 中加入 GeneratorMapper plugin

<!--mybatis代码自动生成插件-->
<build>
    <plugins>
        <plugin>
            <groupId>org.mybatis.generator</groupId>
            <artifactId>mybatis-generator-maven-plugin</artifactId>
            <version>1.3.7</version>
            <configuration>
                <!--配置文件的位置-->
                <configurationFile>GeneratorMapper.xml</configurationFile>
                <verbose>true</verbose>
                <overwrite>true</overwrite>
            </configuration>
        </plugin>
    </plugins>

    <!-- 这三个 resources 使用 IDEA 在编译的时候会出现的问题，所以加上 -->
    <resources>
        <resource>
            <directory>src/main/java</directory>
            <includes>
                <include>**/*.xml</include>
            </includes>
        </resource>
        <resource>
            <directory>src/main/resources</directory>
            <includes>
                <include>**/*.*</include>
            </includes>
        </resource>
        <resource>
            <directory>src/main/webapp</directory>
            <targetPath>META-INF/resources</targetPath>
            <includes>
                <include>**/*.*</include>
            </includes>
        </resource>
    </resources>
</build>

双击执行插件，生成 mapper 和 model，最后能在输出中看到 Build Sucess 并且生成了相关的 mapper 和 model 就成功了。

查询示例

在数据库中插入几条数据，进行查询测试

书写一个 Controller 提供接口，在里面自动导入 Server 层

package me.yanrs.springboot.controller;

import me.yanrs.springboot.service.StudentService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class MybatisController {

    @Autowired
    private StudentService studentService;

    @GetMapping("/boot/students")
    public Object student(){
        return studentService.getAllStudent();
    }
}

创建 Server 层接口，并实现 getAllStudent 方法。

package me.yanrs.springboot.service;

import me.yanrs.springboot.model.Student;

import java.util.List;

public interface StudentService {
    public List<Student> getAllStudent();
}

package me.yanrs.springboot.model.impl;

import me.yanrs.springboot.mapper.StudentMapper;
import me.yanrs.springboot.model.Student;
import me.yanrs.springboot.service.StudentService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class StudentServiceImpl implements StudentService {
    // 这里去和生成的 mapper 接口进行关联，这里需要注意的是，若要 Autowired 成功，需要为 StudentMapper 加上 Mapper 注解
    @Autowired
    private StudentMapper studentMapper;

    @Override
    public List<Student> getAllStudent() {
        return studentMapper.selectAllStudent();
    }
}

注意⚠️：上面代码中若要 Autowired 成功，需要为 StudentMapper 加上 Mapper 注解。

在 StudentMapper 中添加 Mapper 注解，并且新增，selectAllStudent 方法。

package me.yanrs.springboot.mapper;

import me.yanrs.springboot.model.Student;
import org.apache.ibatis.annotations.Mapper;

import java.util.List;

@Mapper
public interface StudentMapper {
    int deleteByPrimaryKey(Integer id);

    int insert(Student record);

    int insertSelective(Student record);

    Student selectByPrimaryKey(Integer id);

    int updateByPrimaryKeySelective(Student record);

    int updateByPrimaryKey(Student record);

    List<Student> selectAllStudent();
}

最后在对应的 mapper xml 文件中创建 selectAllStudent 方法。

<!-- 查询所有学生 -->
<select id="selectAllStudent" resultMap="BaseResultMap">
    select
    <include refid="Base_Column_List"/><!--Base_Column_List 表示查询的是所有字段，在上面定义了 -->
    from student
</select>

运行项目，进行访问 http://127.0.0.1:8080/boot/students 就能看到效果

2. 在访问数据库的 Server 方法上添加注解使用事务
```java
package me.yanrs.springboot.service.impl;

import me.yanrs.springboot.mapper.StudentMapper;
import me.yanrs.springboot.model.Student;
import me.yanrs.springboot.service.StudentService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

@Service
public class StudentServiceImpl implements StudentService {
    // 这里去和生成的 mapper 接口进行关联，这里需要注意的是，若要 Autowired 成功，需要为 StudentMapper 加上 Mapper 注解
    @Autowired
    private StudentMapper studentMapper;

    @Override
    public List<Student> getAllStudent() {
        return studentMapper.selectAllStudent();
    }

    @Transactional // 开启事务
    @Override
    public int update() {
        Student student = new Student();
        student.setId(1);
        student.setName("test1111");
        student.setAge(90);
        int update = studentMapper.updateByPrimaryKeySelective(student);
        System.out.printf("更新影响条数为：" + update);

        // 制作一个运行异常, 因为上面开启了事务，所以上一步的更新就会回滚。如果关闭了上面的事务，那么将不会回滚
        int a = 10/0;

        return update;
    }
}
