ж•°жҚ®еә“иҝҒз§»дё»й”®йҮҚеӨҚй—®йўҳ

еҸ‘иЎЁдәҺ 2018-07-11 | еҲҶзұ»дәҺ еҘҮж·«жҠҖе·§ | жң¬ж–ҮжҖ»йҳ…иҜ»йҮҸ ж¬Ў

еңәжҷҜ

йЎ№зӣ®иҝҒз§»пјҢе°Ҷж•°жҚ®еә“д»Һи…ҫи®Ҝдә‘иҝҒз§»еҲ°йҳҝйҮҢдә‘пјҢйҳҝйҮҢдә‘дёҠзҡ„ж•°жҚ®еә“е·Із»ҸжңүдәҶж•°жҚ®пјҢзҺ°еңЁиҰҒеҒҡзҡ„е°ұжҳҜе°Ҷи…ҫи®Ҝдә‘дёҠзҡ„ж—§ж•°жҚ®иҝҒз§»еҲ°йҳҝйҮҢдә‘дёҠгҖӮйҒҮеҲ°зҡ„й—®йўҳжҳҜдё»й”®йҮҚеӨҚзҡ„й—®йўҳпјҢеӣ дёәж–°ж•°жҚ®еә“дёҠе·Із»ҸжңүдәҶж•°жҚ®пјҢиҖҢиҖҒж•°жҚ®еә“дёҠеҜјеҮәзҡ„ж•°жҚ®д№ҹеҗ«жңүдё»й”®зҡ„дҝЎжҒҜгҖӮ

еҜјеҮәж•°жҚ®

mysqldump -u"з”ЁжҲ·еҗҚ" -p"еҜҶз Ғ" -h"жңҚеҠЎеҷЁең°еқҖ" --databases ж•°жҚ®еә“еҗҚз§° --set-gtid-purged=off --tables иЎЁеҗҚ --no-create-info --log-error=/tmp/sql/mysqldump_error_log.err  | sed -e "s/([0-9]*,/(NULL,/gi" > /tmp/sql/face_feature_1_n.sql

дёҠиҝ°е‘Ҫд»Өзҡ„ж„ҸжҖқжҳҜе°ҶеҜјеҮәжҢҮе®ҡж•°жҚ®еә“дёӢзҡ„жҹҗеј иЎЁзҡ„дҝЎжҒҜпјҢе№¶дё”е°Ҷдё»й”®пјҲз¬¬дёҖеҲ—пјүи®ҫзҪ®дёәNULLгҖӮдҝқеӯҳдёә face_feature_1_n.sql ж–Үд»¶

еҸӮж•°и§ЈйҮҠ

  1. вҖ“set-gtid-purged=off жҳҜеӣ дёәеңЁеҜје…Ҙзҡ„ж—¶еҖҷйҒҮеҲ°й”ҷиҜҜпјҢERROR 1840 (HY000) at line 24: GTID_PURGED can only be set when GTID_EXECUTED is empty.жүҖд»ҘеңЁеҜјеҮәзҡ„ж—¶еҖҷи®ҫзҪ®set-gtid-purged=off
  2. sed ж„ҸдёәжөҒзј–иҫ‘еҷЁпјҲStream EditorпјүпјҢеңЁShellи„ҡжң¬е’ҢMakefileдёӯдҪңдёәиҝҮж»ӨеҷЁдҪҝз”Ёйқһеёёжҷ®йҒҚпјҢжӣҙеӨҡsedе‘Ҫд»ӨпјҢеҸӮиҖғ https://www.cnblogs.com/linux-wangkun/p/5745584.html

еҜје…Ҙж•°жҚ®

mysql -h"жңҚеҠЎеҷЁең°еқҖ" -u"з”ЁжҲ·еҗҚ" -p"еҜҶз Ғ" --default-character-set=utf8 ж•°жҚ®еә“ < face_feature_1_n.sql

е°Ҷface_feature_1_n.sqlдёӯзҡ„ж•°жҚ®пјҢеҜје…ҘеҲ°жҢҮе®ҡзҡ„ж•°жҚ®еә“пјҲеӣ дёәface_feature_1_n.sqlдёӯзҡ„дё»й”®жҳҜдёәNULLпјҢжүҖжңүеңЁжҸ’е…Ҙзҡ„ж—¶еҖҷпјҢдјҡиөӢдәҲж–°зҡ„еҖјпјҢд»ҺиҖҢиҫҫеҲ°дё»й”®дёҚйҮҚеӨҚзҡ„й—®йўҳпјү

еёҰжқҘзҡ„еҪұе“Қ

е…¶е®һиҝҷз§Қж–№жі•д№ҹжҳҜжҠ•жңәеҸ–е·§гҖӮжҒ°еҘҪдё»й”®жҳҜиЎЁзҡ„з¬¬дёҖеҲ—пјҢдҪҶиҝҷж ·д№ҹеҸҜд»Ҙж»Ўи¶іжҲ‘д»¬зҡ„еӨ§йғЁеҲҶйңҖжұӮдәҶ

е…¶д»–ж–№жі•

еӣ дёәжҲ‘д»¬иҝҷйҮҢзҡ„дёҡеҠЎжҳҜе°Ҷж•°жҚ®д»Һи…ҫи®ҜиҝҒз§»еҲ°йҳҝйҮҢпјҢж¶үеҸҠеҲ°ж•°жҚ®еә“зҡ„иҝһйҖҡжҖ§зӯүзӯүй—®йўҳгҖӮд№ӢеүҚд№ҹиҖғиҷ‘иҝҮзӣҙжҺҘеҶҷдёҖдёӘpythonи„ҡжң¬пјҢе°ҶеҺҹжқҘзҡ„ж•°жҚ®еӯҳеӮЁдёәdataframeпјҢз„¶еҗҺеҶҚеҜје…Ҙж–°ж•°жҚ®еә“пјҢдҪҶжҳҜеӣ дёәе…¬еҸёDBAйӮЈиҫ№дёҚиЎҢпјҢжүҖд»ҘеҸӘиғҪиҖҒиҖҒе®һе®һзҡ„еҶҷmysqlзҡ„дёңиҘҝгҖӮдёәдәҶи§ЈеҶіжҲ‘д»¬д»ҘдёҠзҡ„еңәжҷҜж–№жЎҲпјҢдёҚиҖғиҷ‘и·ЁжңҚеҠЎеҷЁзҡ„иҜқпјҢжҲ‘д»¬иҝҳеҸҜд»ҘдҪҝз”Ёд»ҘдёӢж–№жі•

INSERT newly_created_table_name (all, columns, except, the, auto_increment, column)
       SELECT all, columns, except, the, auto_increment, column
         FROM renamed_table_name

йӮЈе°ұжҳҜеңЁжҹҘиҜўзҡ„ж—¶еҖҷпјҢзӣҙжҺҘдёҚжҹҘиҜўidйӮЈдёҖеҲ—пјҢе№¶зӣҙжҺҘе°ҶжҹҘиҜўзҡ„з»“жһңжҸ’е…ҘеҲ°ж–°зҡ„ж•°жҚ®иЎЁдёӯпјҢжҳҫз„¶иҝҷжҳҺжҳҫдёҚз¬ҰеҗҲжҲ‘д»¬зҡ„еңәжҷҜпјҢжүҖд»Ҙе°ұж”ҫејғдәҶгҖӮ
